\begin{algorithm}[!htbp]
{
 \SetAlgoLined
 \DontPrintSemicolon
 \DontPrintSemicolon
 \KwIn{$f$, $\mathit{VM}$, $S$, $M$}
 \KwOut{The optimal configuration}
 $D$ := Initial sampling ($f$, $\mathit{VM}$) \;

 \For{$k$ in $|\mathit{VM} \notin D|$}{
   $p(y|vm, D)$ := FitHighLevelModel(M, D) \;
   $p(y|vm, D, L)$ := FitLowLevelModel(M, D, L) \;
   $vm_k$ := $arg max_{vm \in \mathit{VM}} S(x, p(y|vm, L), p(y|vm, L, D))$ \;
   $y_k, L_k$ := $f(\mathit{vm}_k)$ \;
   $D$ := $D \cup (\mathit{vm}_k, y_k, L_k)$ \;
 }
 \caption{Hybrid Bayesian Optimization}
 \label{alg:hbo}
 }
\end{algorithm}
